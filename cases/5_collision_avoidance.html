A* Pathfinding for Orbital Trajectory
// A* Search Algorithm for Satellite Trajectory Planning
#include <iostream>
#include <vector>
#include <queue>
#include <unordered_map>
#include <cmath>

struct Node3D {
    double x, y, z, time;
    double gCost, hCost;
    Node3D* parent;
    
    double fCost() const { return gCost + hCost; }
};

class SatellitePathPlanner {
private:
    std::vector<std::vector<std::vector<bool>>> obstacleGrid;
    double fuelPenalty;
    
    // Heuristic: Euclidean distance + fuel penalty
    double calculateHeuristic(Node3D* current, Node3D* goal) {
        double distance = sqrt(
            pow(goal->x - current->x, 2) +
            pow(goal->y - current->y, 2) +
            pow(goal->z - current->z, 2)
        );
        return distance + fuelPenalty * abs(goal->time - current->time);
    }
    
    // Check if position is safe from debris
    bool isSafe(int x, int y, int z) {
        return x >= 0 && y >= 0 && z >= 0 &&
               x < obstacleGrid.size() && 
               y < obstacleGrid[0].size() && 
               z < obstacleGrid[0][0].size() &&
               !obstacleGrid[x][y][z];
    }

public:
    std::vector<Node3D*> findOptimalPath(Node3D* start, Node3D* goal) {
        auto compare = [](Node3D* a, Node3D* b) {
            return a->fCost() > b->fCost();
        };
        
        std::priority_queue<Node3D*, std::vector<Node3D*>, decltype(compare)> openSet(compare);
        std::unordered_map<Node3D*, bool> inOpenSet;
        
        start->gCost = 0;
        start->hCost = calculateHeuristic(start, goal);
        openSet.push(start);
        inOpenSet[start] = true;
        
        while (!openSet.empty()) {
            Node3D* current = openSet.top();
            openSet.pop();
            
            if (current == goal) {
                // Reconstruct path
                std::vector<Node3D*> path;
                while (current != nullptr) {
                    path.push_back(current);
                    current = current->parent;
                }
                std::reverse(path.begin(), path.end());
                return path;
            }
            
            // Explore 26 possible 3D movements
            for (int dx = -1; dx <= 1; dx++) {
                for (int dy = -1; dy <= 1; dy++) {
                    for (int dz = -1; dz <= 1; dz++) {
                        if (dx == 0 && dy == 0 && dz == 0) continue;
                        
                        int newX = current->x + dx;
                        int newY = current->y + dy;
                        int newZ = current->z + dz;
                        
                        if (isSafe(newX, newY, newZ)) {
                            Node3D* neighbor = new Node3D{newX, newY, newZ, current->time + 1};
                            double tentativeG = current->gCost + sqrt(dx*dx + dy*dy + dz*dz);
                            
                            if (tentativeG < neighbor->gCost) {
                                neighbor->parent = current;
                                neighbor->gCost = tentativeG;
                                neighbor->hCost = calculateHeuristic(neighbor, goal);
                                
                                if (inOpenSet.find(neighbor) == inOpenSet.end()) {
                                    openSet.push(neighbor);
                                    inOpenSet[neighbor] = true;
                                }
                            }
                        }
                    }
                }
            }
        }
        
        return {}; // No path found
    }
};
Segment Tree for Collision Window Detection
// Segment Tree for Fast Collision Window Queries
class CollisionSegmentTree {
private:
    std::vector<int> tree;
    int n;
    
    void update(int node, int start, int end, int idx, int value) {
        if (start == end) {
            tree[node] = value;
        } else {
            int mid = (start + end) / 2;
            if (idx <= mid)
                update(2 * node, start, mid, idx, value);
            else
                update(2 * node + 1, mid + 1, end, idx, value);
            
            tree[node] = std::max(tree[2 * node], tree[2 * node + 1]);
        }
    }
    
    int queryRange(int node, int start, int end, int l, int r) {
        if (r < start || end < l)
            return 0;
        
        if (l <= start && end <= r)
            return tree[node];
        
        int mid = (start + end) / 2;
        int p1 = queryRange(2 * node, start, mid, l, r);
        int p2 = queryRange(2 * node + 1, mid + 1, end, l, r);
        return std::max(p1, p2);
    }

public:
    CollisionSegmentTree(int size) : n(size) {
        tree.resize(4 * n);
    }
    
    // Update collision risk at time t
    void updateCollisionRisk(int timeIndex, int riskLevel) {
        update(1, 0, n - 1, timeIndex, riskLevel);
    }
    
    // Query maximum collision risk in time window [t1, t2]
    int getMaxRiskInWindow(int t1, int t2) {
        return queryRange(1, 0, n - 1, t1, t2);
    }
};

// Usage Example
int main() {
    CollisionSegmentTree collisionTracker(10000); // 10000 time units
    
    // Update collision risks for different time windows
    collisionTracker.updateCollisionRisk(1500, 8); // High risk at t=1500
    collisionTracker.updateCollisionRisk(2300, 3); // Medium risk at t=2300
    
    // Query risk in time window
    int maxRisk = collisionTracker.getMaxRiskInWindow(1400, 1600);
    std::cout << "Maximum collision risk: " << maxRisk << std::endl;
    
    return 0;
}
2D Kadane's Algorithm for Debris Field Analysis
// 2D Kadane's Algorithm for Maximum Debris Density Detection
#include <vector>
#include <algorithm>
#include <climits>

class DebrisFieldAnalyzer {
private:
    // 1D Kadane's algorithm for maximum subarray
    int kadane1D(std::vector<int>& arr) {
        int maxSum = INT_MIN;
        int currentSum = 0;
        
        for (int i = 0; i < arr.size(); i++) {
            currentSum += arr[i];
            maxSum = std::max(maxSum, currentSum);
            
            if (currentSum < 0) {
                currentSum = 0;
            }
        }
        
        return maxSum;
    }

public:
    // Find maximum debris density in 2D orbital space
    int findMaxDebrisDensity(std::vector<std::vector<int>>& debrisGrid) {
        int rows = debrisGrid.size();
        int cols = debrisGrid[0].size();
        int maxDensity = INT_MIN;
        
        // Try all possible top and bottom row combinations
        for (int top = 0; top < rows; top++) {
            std::vector<int> tempSum(cols, 0);
            
            for (int bottom = top; bottom < rows; bottom++) {
                // Add current row to temp array
                for (int col = 0; col < cols; col++) {
                    tempSum[col] += debrisGrid[bottom][col];
                }
                
                // Apply 1D Kadane's to find max sum in this row range
                int currentMax = kadane1D(tempSum);
                maxDensity = std::max(maxDensity, currentMax);
            }
        }
        
        return maxDensity;
    }
    
    // Find coordinates of maximum debris density region
    struct DebrisRegion {
        int topRow, bottomRow, leftCol, rightCol, maxDensity;
    };
    
    DebrisRegion findMaxDebrisRegion(std::vector<std::vector<int>>& debrisGrid) {
        int rows = debrisGrid.size();
        int cols = debrisGrid[0].size();
        DebrisRegion result = {0, 0, 0, 0, INT_MIN};
        
        for (int top = 0; top < rows; top++) {
            std::vector<int> tempSum(cols, 0);
            
            for (int bottom = top; bottom < rows; bottom++) {
                for (int col = 0; col < cols; col++) {
                    tempSum[col] += debrisGrid[bottom][col];
                }
                
                // Modified Kadane's to track indices
                int maxSum = INT_MIN, currentSum = 0;
                int start = 0, end = 0, tempStart = 0;
                
                for (int i = 0; i < cols; i++) {
                    currentSum += tempSum[i];
                    
                    if (currentSum > maxSum) {
                        maxSum = currentSum;
                        start = tempStart;
                        end = i;
                    }
                    
                    if (currentSum < 0) {
                        currentSum = 0;
                        tempStart = i + 1;
                    }
                }
                
                if (maxSum > result.maxDensity) {
                    result = {top, bottom, start, end, maxSum};
                }
            }
        }
        
        return result;
    }
};
