C++ - Segment Tree & Fenwick Tree 
// SpaceX Launch Scheduling System
#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

class SegmentTree {
private:
    vector<int> tree;
    int n;
    
    void build(int node, int start, int end, vector<int>& arr) {
        if (start == end) {
            tree[node] = arr[start];
        } else {
            int mid = (start + end) / 2;
            build(2 * node, start, mid, arr);
            build(2 * node + 1, mid + 1, end, arr);
            tree[node] = max(tree[2 * node], tree[2 * node + 1]);
        }
    }
    
    int query(int node, int start, int end, int l, int r) {
        if (r < start || end < l) return 0;
        if (l <= start && end <= r) return tree[node];
        
        int mid = (start + end) / 2;
        int p1 = query(2 * node, start, mid, l, r);
        int p2 = query(2 * node + 1, mid + 1, end, l, r);
        return max(p1, p2);
    }
    
public:
    SegmentTree(vector<int>& arr) {
        n = arr.size();
        tree.resize(4 * n);
        build(1, 0, n - 1, arr);
    }
    
    int rangeMaxQuery(int l, int r) {
        return query(1, 0, n - 1, l, r);
    }
};

class FenwickTree {
private:
    vector<int> bit;
    int n;
    
public:
    FenwickTree(int n) : n(n), bit(n + 1, 0) {}
    
    void update(int idx, int delta) {
        for (++idx; idx <= n; idx += idx & -idx)
            bit[idx] += delta;
    }
    
    int query(int idx) {
        int sum = 0;
        for (++idx; idx > 0; idx -= idx & -idx)
            sum += bit[idx];
        return sum;
    }
    
    int rangeQuery(int l, int r) {
        return query(r) - query(l - 1);
    }
};

class LaunchScheduler {
private:
    SegmentTree* segTree;
    FenwickTree* fenTree;
    vector<int> padAvailability;
    vector<int> resourceUsage;
    
public:
    LaunchScheduler(int timeSlots) {
        padAvailability.resize(timeSlots, 100); // 100% available initially
        segTree = new SegmentTree(padAvailability);
        fenTree = new FenwickTree(timeSlots);
    }
    
    bool checkAvailability(int startTime, int endTime, int requiredCapacity) {
        int maxUsage = segTree->rangeMaxQuery(startTime, endTime);
        int totalUsage = fenTree->rangeQuery(startTime, endTime);
        
        cout << "🔍 Checking slot [" << startTime << ", " << endTime << "]" << endl;
        cout << "   Max Usage: " << maxUsage << "%" << endl;
        cout << "   Total Usage: " << totalUsage << "%" << endl;
        
        return (maxUsage + requiredCapacity <= 100);
    }
    
    void scheduleLaunch(int startTime, int endTime, int capacity) {
        if (checkAvailability(startTime, endTime, capacity)) {
            for (int i = startTime; i <= endTime; i++) {
                fenTree->update(i, capacity);
            }
            cout << "✅ Launch scheduled successfully!" << endl;
            cout << "🚀 Time: " << startTime << ":00 - " << endTime << ":00" << endl;
            cout << "⛽ Capacity: " << capacity << "%" << endl;
        } else {
            cout << "❌ Insufficient capacity for launch!" << endl;
        }
        cout << "----------------------------" << endl;
    }
    
    void displayStatus() {
        cout << "📊 Current Launch Schedule Status:" << endl;
        for (int i = 0; i < 24; i++) {
            int usage = fenTree->query(i) - (i > 0 ? fenTree->query(i-1) : 0);
            cout << "Hour " << i << ": " << usage << "% used" << endl;
        }
    }
};

int main() {
    cout << "🚀 SpaceX Launch Scheduling System Initialized" << endl;
    cout << "================================================" << endl;
    
    LaunchScheduler scheduler(24); // 24-hour scheduling
    
    // Simulate launch requests
    scheduler.scheduleLaunch(8, 10, 60);  // Starlink mission
    scheduler.scheduleLaunch(9, 11, 50);  // NASA mission (overlap test)
    scheduler.scheduleLaunch(14, 16, 40); // Commercial mission
    scheduler.scheduleLaunch(20, 22, 70); // DoD mission
    
    scheduler.displayStatus();
    
    return 0;
}

C++ - Weather Integration & Priority Queue 
// Advanced Launch Scheduling with Weather and Priority Management
#include <queue>
#include <unordered_map>
#include <ctime>

struct LaunchMission {
    int id;
    int priority;
    int startTime;
    int duration;
    int resourceReq;
    string missionType;
    
    bool operator<(const LaunchMission& other) const {
        return priority < other.priority; // Max heap for priority
    }
};

class WeatherSystem {
private:
    unordered_map<int, double> weatherScore; // Hour -> Weather score (0-1)
    
public:
    WeatherSystem() {
        // Initialize with sample weather data
        srand(time(nullptr));
        for (int i = 0; i < 24; i++) {
            weatherScore[i] = 0.3 + (double)rand() / RAND_MAX * 0.7;
        }
    }
    
    double getWeatherScore(int hour) {
        return weatherScore[hour];
    }
    
    bool isLaunchable(int startHour, int duration) {
        for (int i = startHour; i < startHour + duration; i++) {
            if (weatherScore[i % 24] < 0.6) return false;
        }
        return true;
    }
};

class AdvancedLaunchScheduler {
private:
    priority_queue<LaunchMission> missionQueue;
    LaunchScheduler* baseScheduler;
    WeatherSystem* weather;
    vector<LaunchMission> scheduledMissions;
    
public:
    AdvancedLaunchScheduler() {
        baseScheduler = new LaunchScheduler(24);
        weather = new WeatherSystem();
    }
    
    void addMission(int id, int priority, int startTime, 
                   int duration, int resourceReq, string type) {
        LaunchMission mission = {id, priority, startTime, duration, resourceReq, type};
        missionQueue.push(mission);
        
        cout << "🆕 Mission Added: " << type << " (Priority: " << priority << ")" << endl;
    }
    
    void processMissions() {
        cout << "🔄 Processing missions by priority..." << endl;
        cout << "========================================" << endl;
        
        while (!missionQueue.empty()) {
            LaunchMission mission = missionQueue.top();
            missionQueue.pop();
            
            cout << "🎯 Processing: " << mission.missionType 
                 << " (ID: " << mission.id << ")" << endl;
            
            // Check weather conditions
            if (!weather->isLaunchable(mission.startTime, mission.duration)) {
                cout << "🌦️  Weather conditions unfavorable, finding alternative..." << endl;
                mission.startTime = findAlternativeSlot(mission);
            }
            
            // Check resource availability
            if (baseScheduler->checkAvailability(mission.startTime, 
                                               mission.startTime + mission.duration, 
                                               mission.resourceReq)) {
                baseScheduler->scheduleLaunch(mission.startTime, 
                                            mission.startTime + mission.duration, 
                                            mission.resourceReq);
                scheduledMissions.push_back(mission);
                
                cout << "✅ " << mission.missionType << " scheduled successfully!" << endl;
                cout << "   📅 Time: " << mission.startTime << ":00 - " 
                     << (mission.startTime + mission.duration) << ":00" << endl;
                cout << "   🌤️  Weather Score: " 
                     << weather->getWeatherScore(mission.startTime) * 100 << "%" << endl;
            } else {
                cout << "❌ Mission " << mission.missionType << " could not be scheduled" << endl;
            }
            cout << "----------------------------------------" << endl;
        }
    }
    
    int findAlternativeSlot(LaunchMission& mission) {
        // Find best alternative slot considering weather and resources
        int bestSlot = mission.startTime;
        double bestScore = 0;
        
        for (int hour = 0; hour < 24; hour++) {
            if (weather->isLaunchable(hour, mission.duration)) {
                double weatherScore = weather->getWeatherScore(hour);
                if (weatherScore > bestScore) {
                    bestScore = weatherScore;
                    bestSlot = hour;
                }
            }
        }
        
        cout << "🔄 Alternative slot found: " << bestSlot << ":00" << endl;
        return bestSlot;
    }
    
    void displaySchedule() {
        cout << "📋 Final Launch Schedule:" << endl;
        cout << "=========================" << endl;
        
        for (const auto& mission : scheduledMissions) {
            cout << "🚀 " << mission.missionType << " | Time: " 
                 << mission.startTime << ":00-" 
                 << (mission.startTime + mission.duration) << ":00"
                 << " | Priority: " << mission.priority << endl;
        }
    }
};

int main() {
    cout << "🚀 Advanced SpaceX Launch Scheduling System" << endl;
    cout << "=============================================" << endl;
    
    AdvancedLaunchScheduler scheduler;
    
    // Add various missions with different priorities
    scheduler.addMission(1, 10, 8, 2, 60, "🛰️ Starlink");
    scheduler.addMission(2, 15, 10, 3, 70, "🚀 NASA Artemis");
    scheduler.addMission(3, 12, 14, 2, 50, "🛡️ DoD Mission");
    scheduler.addMission(4, 8, 16, 1, 30, "💼 Commercial");
    scheduler.addMission(5, 20, 6, 4, 80, "🌙 Lunar Gateway");
    
    scheduler.processMissions();
    scheduler.displaySchedule();
    
    cout << "\n🎉 Scheduling complete! System optimized for maximum efficiency." << endl;
    
    return 0;
}
