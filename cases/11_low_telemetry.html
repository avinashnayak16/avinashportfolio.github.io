🧠 Data Structures & Algorithms

Persistent Segment Tree

Purpose: Query historical telemetry versions without rebuilding

Time Complexity: O(log n) query, O(n log n) build

Use Case: "Max thrust from T+100 to T+400s across 5 launches"

Mo's Algorithm

Purpose: Efficient offline range queries for bulk analysis

Time Complexity: O((n + q) × √n)

Use Case: Vibration pattern analysis across multiple flights

Fenwick Tree

Purpose: Real-time cumulative updates

Time Complexity: O(log n) update/query

Use Case: Live temperature averages and trend analysis

Priority Queue

Purpose: Alert system for critical anomalies

Time Complexity: O(log n) insert/extract

Use Case: Overheating alerts, abnormal thrust detection

Persistent Segment Tree Implementation 
// Persistent Segment Tree for SpaceX Telemetry Data
#include <iostream>
#include <vector>
#include <memory>

struct Node {
    double maxValue;
    std::shared_ptr<Node> left, right;
    
    Node(double val = 0) : maxValue(val), left(nullptr), right(nullptr) {}
};

class PersistentSegmentTree {
private:
    std::vector<std::shared_ptr<Node>> roots;
    int n;
    
    std::shared_ptr<Node> build(const std::vector<double>& arr, int l, int r) {
        if (l == r) {
            return std::make_shared<Node>(arr[l]);
        }
        
        auto node = std::make_shared<Node>();
        int mid = (l + r) / 2;
        node->left = build(arr, l, mid);
        node->right = build(arr, mid + 1, r);
        node->maxValue = std::max(node->left->maxValue, node->right->maxValue);
        
        return node;
    }
    
    std::shared_ptr<Node> update(std::shared_ptr<Node> prev, int l, int r, int pos, double val) {
        if (l == r) {
            return std::make_shared<Node>(val);
        }
        
        auto node = std::make_shared<Node>();
        int mid = (l + r) / 2;
        
        if (pos <= mid) {
            node->left = update(prev->left, l, mid, pos, val);
            node->right = prev->right;
        } else {
            node->left = prev->left;
            node->right = update(prev->right, mid + 1, r, pos, val);
        }
        
        node->maxValue = std::max(node->left->maxValue, node->right->maxValue);
        return node;
    }
    
    double query(std::shared_ptr<Node> node, int l, int r, int ql, int qr) {
        if (ql > r || qr < l) return -1e9;
        if (ql <= l && r <= qr) return node->maxValue;
        
        int mid = (l + r) / 2;
        return std::max(
            query(node->left, l, mid, ql, qr),
            query(node->right, mid + 1, r, ql, qr)
        );
    }

public:
    PersistentSegmentTree(const std::vector<double>& telemetryData) {
        n = telemetryData.size();
        roots.push_back(build(telemetryData, 0, n - 1));
    }
    
    // Create new version with updated telemetry value
    void updateTelemetry(int timeIndex, double newValue) {
        auto newRoot = update(roots.back(), 0, n - 1, timeIndex, newValue);
        roots.push_back(newRoot);
    }
    
    // Query max value in time range for specific version
    double getMaxInRange(int version, int startTime, int endTime) {
        return query(roots[version], 0, n - 1, startTime, endTime);
    }
    
    int getVersionCount() { return roots.size(); }
};
Mo's Algorithm for Batch Queries 
// Mo's Algorithm for Efficient Batch Telemetry Analysis
#include <algorithm>
#include <cmath>

struct Query {
    int left, right, index;
    double result;
};

class TelemetryAnalyzer {
private:
    std::vector<double> data;
    int blockSize;
    double currentMax;
    std::multiset<double> currentWindow;
    
    void add(int pos) {
        currentWindow.insert(data[pos]);
        currentMax = *currentWindow.rbegin();
    }
    
    void remove(int pos) {
        currentWindow.erase(currentWindow.find(data[pos]));
        currentMax = currentWindow.empty() ? 0 : *currentWindow.rbegin();
    }

public:
    TelemetryAnalyzer(const std::vector<double>& telemetryData) 
        : data(telemetryData), currentMax(0) {
        blockSize = sqrt(data.size()) + 1;
    }
    
    std::vector<double> processBatchQueries(std::vector<Query>& queries) {
        // Sort queries by Mo's algorithm ordering
        sort(queries.begin(), queries.end(), [this](const Query& a, const Query& b) {
            int blockA = a.left / blockSize;
            int blockB = b.left / blockSize;
            if (blockA != blockB) return blockA < blockB;
            return (blockA & 1) ? a.right < b.right : a.right > b.right;
        });
        
        int currL = 0, currR = -1;
        
        for (auto& query : queries) {
            int L = query.left, R = query.right;
            
            // Extend/shrink current window to match query range
            while (currR < R) add(++currR);
            while (currL > L) add(--currL);
            while (currR > R) remove(currR--);
            while (currL < L) remove(currL++);
            
            query.result = currentMax;
        }
        
        // Sort back by original query index
        sort(queries.begin(), queries.end(), 
             [](const Query& a, const Query& b) { return a.index < b.index; });
        
        std::vector<double> results;
        for (const auto& q : queries) {
            results.push_back(q.result);
        }
        return results;
    }
};

// SpaceX Mission Control Telemetry System
#include <chrono>
#include <thread>

class SpaceXTelemetrySystem {
private:
    PersistentSegmentTree thrustData;
    PersistentSegmentTree temperatureData;
    TelemetryAnalyzer vibrationAnalyzer;
    
public:
    SpaceXTelemetrySystem(const std::vector<double>& thrust, 
                           const std::vector<double>& temp,
                           const std::vector<double>& vibration)
        : thrustData(thrust), temperatureData(temp), vibrationAnalyzer(vibration) {}
    
    // Real-time telemetry update from rocket
    void updateLiveTelemetry(int timeStep, double thrust, double temp) {
        thrustData.updateTelemetry(timeStep, thrust);
        temperatureData.updateTelemetry(timeStep, temp);
        
        // Alert system for anomalies
        if (thrust > 9000 || temp > 800) {
            std::cout << "🚨 ALERT: Anomaly detected at T+" << timeStep 
                      << "s - Thrust: " << thrust << " temp: " << temp << std::endl;
        }
    }
    
    // Historical analysis: "Max thrust between T+100 to T+400s during last 5 launches"
    std::vector<double> analyzeHistoricalPerformance(int startTime, int endTime, int numVersions = 5) {
        std::vector<double> results;
        int totalVersions = thrustData.getVersionCount();
        
        for (int i = std::max(0, totalVersions - numVersions); i < totalVersions; i++) {
            double maxThrust = thrustData.getMaxInRange(i, startTime, endTime);
            results.push_back(maxThrust);
        }
        
        return results;
    }
    
    // Mission critical: Stage separation analysis
    void analyzeStageSeparation(int separationTime) {
        auto start = std::chrono::high_resolution_clock::now();
        
        // Query thrust 10s before and after separation
        int currentVersion = thrustData.getVersionCount() - 1;
        double preThrust = thrustData.getMaxInRange(currentVersion, 
                                                     separationTime - 10, separationTime);
        double postThrust = thrustData.getMaxInRange(currentVersion, 
                                                      separationTime, separationTime + 10);
        
        auto end = std::chrono::high_resolution_clock::now();
        auto duration = std::chrono::duration_cast<std::chrono::microseconds>(end - start);
        
        std::cout << "🚀 Stage Separation Analysis (T+" << separationTime << "s):\n"
                  << "   Pre-separation max thrust: " << preThrust << " kN\n"
                  << "   Post-separation max thrust: " << postThrust << " kN\n"
                  << "   Query executed in: " << duration.count() << " μs\n" << std::endl;
    }
    
    // Batch analysis using Mo's algorithm
    void runBatchVibrationAnalysis() {
        std::cout << "🔍 Running batch vibration analysis across multiple flights...\n";
        
        std::vector<Query> queries = {
            {50, 150, 0, 0},   // T+50 to T+150s
            {200, 300, 1, 0}, // T+200 to T+300s (Max-Q phase)
            {400, 500, 2, 0}, // T+400 to T+500s (Second stage)
        };
        
        auto results = vibrationAnalyzer.processBatchQueries(queries);
        
        for (size_t i = 0; i < results.size(); i++) {
            std::cout << "   Query " << i + 1 << ": Max vibration = " 
                      << results[i] << " Hz\n";
        }
    }
};

int main() {
    // Simulated telemetry data from Falcon 9 launch
